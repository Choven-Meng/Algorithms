实现组合算法C(n,k)，可以用递归实现：
```
def combine(lst, l):
    result = []
    tmp = [0] * l
    length = len(lst)

    def next_num(li=0, ni=0):
        if ni == l:
            result.append(tmp.copy())
            return
        for lj in range(li, length):
            tmp[ni] = lst[lj]
            # print(tmp)
            next_num(lj + 1, ni + 1)

    next_num()
    return result

lst  = [2,5,3,4]
print(combine(lst,2))

#output:
[[2, 5], [2, 3], [2, 4], [5, 3], [5, 4], [3, 4]]
```

实现排列算法A(n,k)，用递归实现：
```
import copy

def permutation(lst,k):
    result = []
    length = len(lst)
    tmp = [0]*k
    def next_num(a,ni=0):
        if ni == k:
            result.append(copy.copy(tmp))
            return
        for lj in a:
            tmp[ni] = lj
            b = a[:]
            b.pop(a.index(lj))
            next_num(b,ni+1)
    c = lst[:]
    next_num(c,0)
    return result

lst  = [2,5,3,4]
print(permutation(lst,2))

#output:
[[2, 5], [2, 3], [2, 4], [5, 2], [5, 3], [5, 4], [3, 2], [3, 5], [3, 4], [4, 2], [4, 5], [4, 3]]
```
